import math

def polylog(x):
soma=0
for k in range(1,10000):
soma=soma+x**k/k**2
return soma

def f (x):
resu=math.log(1-x)*polylog(1-x**0.5)/x
return resu


a = 0.0001
b = 1


n = 1_000_000
dx = (b - a) / n

integ = 0.5 * (f(a) + f(b))
for i in range(1, n):
x = a + i * dx
integ += f(x)
integ *= dx

print(integ)
